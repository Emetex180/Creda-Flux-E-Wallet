from flask import Flask, render_template, request, jsonify from flask_cors import CORS from werkzeug.utils import secure_filename import os app = Flask(__name__) CORS(app) # ========================= # CONFIG # ========================= UPLOAD_FOLDER = "uploads" app.config["UPLOAD_FOLDER"] = UPLOAD_FOLDER ALLOWED_EXTENSIONS = { "png", "jpg", "jpeg", "svg" } MAX_FILE_SIZE = 5 * 1024 * 1024 # 5MB os.makedirs(UPLOAD_FOLDER, exist_ok=True) # ========================= # HELPERS # ========================= def allowed_file(filename): return ( "." in filename and filename.rsplit(".", 1)[1].lower() in ALLOWED_EXTENSIONS ) # ========================= # PAGES # ========================= @app.route("/") def home(): return render_template("register.html") @app.route("/login") def login_page(): return render_template("login.html") # ========================= # REGISTER API # ========================= @app.route("/api/register", methods=["POST"]) def register(): try: # FORM DATA full_legal_name = request.form.get("fullLegalName") email = request.form.get("email") mobile_number = request.form.get("mobileNumber") legal_gender = request.form.get("legalGender") date_of_birth = request.form.get("dateOfBirth") primary_occupation = request.form.get("primaryOccupation") place_of_birth = request.form.get("placeOfBirth") current_residence = request.form.get("currentResidence") portfolio_type = request.form.get("portfolioType") base_currency = request.form.get("baseCurrency") master_password = request.form.get("masterPassword") app_pin = request.form.get("appPin") # FILE uploaded_file = request.files.get("governmentId") filename = None # VALIDATION if not full_legal_name or not email: return jsonify({ "success": False, "message": "Required fields missing" }), 400 # FILE VALIDATION if uploaded_file and uploaded_file.filename != "": if not allowed_file(uploaded_file.filename): return jsonify({ "success": False, "message": "Invalid file type" }), 400 # CHECK FILE SIZE uploaded_file.seek(0, os.SEEK_END) file_size = uploaded_file.tell() uploaded_file.seek(0) if file_size > MAX_FILE_SIZE: return jsonify({ "success": False, "message": "File exceeds 5MB limit" }), 400 # SAVE FILE filename = secure_filename(uploaded_file.filename) save_path = os.path.join( app.config["UPLOAD_FOLDER"], filename ) uploaded_file.save(save_path) # PRINT DATA print({ "full_legal_name": full_legal_name, "email": email, "mobile_number": mobile_number, "legal_gender": legal_gender, "date_of_birth": date_of_birth, "primary_occupation": primary_occupation, "place_of_birth": place_of_birth, "current_residence": current_residence, "portfolio_type": portfolio_type, "base_currency": base_currency, "master_password": master_password, "app_pin": app_pin, "uploaded_file": filename }) return jsonify({ "success": True, "message": "Vault Initialized Successfully" }) except Exception as e: print(e) return jsonify({ "success": False, "message": "Server Error" }), 500 # ========================= # LOGIN API # ========================= @app.route("/api/login", methods=["POST"]) def login(): try: data = request.get_json() email = data.get("email") password = data.get("password") # DEMO LOGIN if ( email == "admin@credaflux.online" and password == "123456" ): return jsonify({ "success": True, "message": "Login Successful" }) return jsonify({ "success": False, "message": "Invalid Credentials" }), 401 except Exception as e: print(e) return jsonify({ "success": False, "message": "Login Failed" }), 500 # ========================= # START SERVER # ========================= if __name__ == "__main__": app.run( debug=True, host="0.0.0.0", port=5000 )
